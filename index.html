Install kibana3

node /box/ {
  include kibana3
}

Configure kibana3

node /box/ {
  class { 'kibana3':
    elasticsearch_host  => '127.0.0.1',
    elasticsearch_index => 'kibana-int',
    elasticsearch_port  => '9200',
  }
}
